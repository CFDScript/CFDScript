Usage Example

Here some example code demonstrating how to use the library to solve a 2D Laplace equation. 

// Definitions
let u = []; // Initialize x as an empty array
const nex = 5; // Number of elements in x-direction
const ney = 5; // Number of elements in y-direction
const xlast = 1; // Last x-coordinate
const ylast = 1; // Last y-coordinate
const boundaryConditions = {
  neumannTop: true,
  neumannBottom: false,
  neumannLeft: false,
  neumannRight: false,
  dirichletTop: false,
  dirichletBottom: true,
  dirichletLeft: true,
  dirichletRight: true,
  dirichletValueTop: 0.0,
  dirichletValueBottom: 1.0,
  dirichletValueLeft: 1.0,
  dirichletValueRight: 1.0,
};
    

import { CFDScript } from './scripts/CFDScript.js';
import { plotSol2D } from './scripts/plotSol.js';

// The functions will be executed as soon as the page finishes loading
window.addEventListener('DOMContentLoaded', (event) => {
  // Call the cfdscript function and retrieve the solution
  let { u, nx, ny, axpt, aypt } = CFDScript(nex, ney, xlast, ylast, boundaryConditions);
});
